    <script>
        let cart = [];

        function addToCart(name, price) {
            cart.push({name, price});
            document.getElementById('cart-count').textContent = cart.length;
            toggleCart();
        }

        function toggleCart() {
            document.getElementById('cart-sidebar').classList.toggle('hidden');
            document.getElementById('cart-overlay').classList.toggle('hidden');
            renderCart();
        }

        function renderCart() {
            const container = document.getElementById('cart-items');
            container.innerHTML = '';
            let total = 0;
            if (cart.length === 0) {
                container.innerHTML = '<p class="text-center text-gray-500 py-12">Carrito vacío</p>';
                document.getElementById('cart-total').textContent = '€0';
                return;
            }
            cart.forEach((item, i) => {
                total += item.price;
                container.innerHTML += `
                    <div class="flex justify-between bg-gray-100 p-4 rounded-xl mb-3">
                        <div><p>${item.name}</p><p class="text-sm text-gray-500">€${item.price}</p></div>
                        <button onclick="removeFromCart(${i})" class="text-red-500">Eliminar</button>
                    </div>`;
            });
            document.getElementById('cart-total').textContent = `€${total}`;
        }

        function removeFromCart(i) {
            cart.splice(i, 1);
            document.getElementById('cart-count').textContent = cart.length;
            renderCart();
        }

        function proceedToPayment() {
            if (cart.length === 0) return;
            let total = cart.reduce((sum, item) => sum + item.price, 0);
            let msg = `Hola, quiero comprar:\n\n`;
            cart.forEach(item => msg += `- ${item.name} → €${item.price}\n`);
            msg += `\nTotal: €${total}\n\nQuiero agendar mi sesión. ¿Disponibilidad?`;
            window.open(`https://wa.me/353876071392?text=${encodeURIComponent(msg)}`, '_blank');
            setTimeout(() => { cart = []; document.getElementById('cart-count').textContent = '0'; toggleCart(); }, 1500);
        }

        window.onload = () => {
            document.getElementById('cart-count').textContent = '0';
            console.log("✅ Página cargada");
        };
    </script>
